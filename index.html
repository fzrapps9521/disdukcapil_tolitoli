Tautan Web App Apps Script

Tempel URL Web App dari Google Apps Script yang telah Anda deploy di bawah ini untuk menghubungkan aplikasi ke Database Google Sheets milik Anda.

1 Langkah Setup di Google Drive

  1. Buat Google Spreadsheet baru di akun Google/Google Workspace instansi Anda.
  2. Pilih menu Ekstensi > Apps Script.
  3. Hapus semua kode bawaan, salin dan tempel kode di bawah ini:
Code.gs 
/* ================================================================================
  GOOGLE APPS SCRIPT BACKEND FOR DISDUKCAPIL (UPDATED)
  - Penambahan kolom logo_url dan carousel_urls di Settings
  - Support Role Admin & Editor
================================================================================
*/

const SPREADSHEET_ID = ""; // Kosongkan jika ingin menggunakan file Spreadsheet ini

function setupDatabase() {
  const ss = SPREADSHEET_ID ? SpreadsheetApp.openById(SPREADSHEET_ID) : SpreadsheetApp.getActiveSpreadsheet();
  
  // Sheet Users
  let sheetUsers = ss.getSheetByName("Users");
  if (!sheetUsers) {
    sheetUsers = ss.insertSheet("Users");
    sheetUsers.appendRow(["id", "email", "password_hash", "name", "role", "created_at"]);
    sheetUsers.appendRow(["admin-1", "admin@disdukcapil.com", "admin123", "Kepala TI", "admin", new Date().toISOString()]);
    sheetUsers.appendRow(["editor-1", "editor@disdukcapil.com", "editor123", "Staf Penulis", "editor", new Date().toISOString()]);
  }
  
  // Sheet Posts
  let sheetPosts = ss.getSheetByName("Posts");
  if (!sheetPosts) {
    sheetPosts = ss.insertSheet("Posts");
    sheetPosts.appendRow(["id", "title", "slug", "content", "thumbnail_url", "category_id", "author_id", "status", "created_at", "updated_at"]);
  }
  
  // Sheet Categories
  let sheetCategories = ss.getSheetByName("Categories");
  if (!sheetCategories) {
    sheetCategories = ss.insertSheet("Categories");
    sheetCategories.appendRow(["id", "name", "slug"]);
    sheetCategories.appendRow(["cat-1", "Pengumuman", "pengumuman"]);
    sheetCategories.appendRow(["cat-2", "Berita", "berita"]);
  }
  
  // Sheet Settings
  let sheetSettings = ss.getSheetByName("Settings");
  if (!sheetSettings) {
    sheetSettings = ss.insertSheet("Settings");
    sheetSettings.appendRow(["key", "value"]);
    sheetSettings.appendRow(["site_title", "DISDUKCAPIL Tolitoli"]);
    sheetSettings.appendRow(["logo_url", ""]);
    sheetSettings.appendRow(["carousel_urls", "https://images.unsplash.com/photo-1541872703-74c5e44368f9?q=80&w=1200\nhttps://images.unsplash.com/photo-1509062522246-3755977927d7?q=80&w=1200"]);
    sheetSettings.appendRow(["contact_email", "info@disdukcapil.tolitolikab.go.id"]);
    sheetSettings.appendRow(["office_address", "Jl. Jenderal Sudirman No. 12, Kab. Tolitoli"]);
    sheetSettings.appendRow(["phone", "0811-1234-5678"]);
    sheetSettings.appendRow(["copyright", "© 2026 DISDUKCAPIL Kab. Tolitoli."]);
  }
  Logger.log("DATABASE BERHASIL DIINISIALISASI!");
}

function doGet(e) {
  const action = e.parameter.action;
  const ss = SPREADSHEET_ID ? SpreadsheetApp.openById(SPREADSHEET_ID) : SpreadsheetApp.getActiveSpreadsheet();
  try {
    if (action === "getPublicData") {
      return contentResponse({
        posts: getSheetData(ss.getSheetByName("Posts")),
        categories: getSheetData(ss.getSheetByName("Categories")),
        settings: getSettingsMap(ss.getSheetByName("Settings")),
        users: getSheetData(ss.getSheetByName("Users")).map(u => ({ id: u.id, name: u.name, role: u.role }))
      });
    } else if (action === "getAdminData") {
      return contentResponse({
        posts: getSheetData(ss.getSheetByName("Posts")),
        categories: getSheetData(ss.getSheetByName("Categories")),
        settings: getSettingsMap(ss.getSheetByName("Settings")),
        users: getSheetData(ss.getSheetByName("Users")).map(u => ({ id: u.id, email: u.email, name: u.name, role: u.role }))
      });
    }
  } catch (err) {
    return contentResponse({ status: "error", message: err.toString() });
  }
}

function doPost(e) {
  let postData;
  try { postData = JSON.parse(e.postData.contents); } catch (err) { return contentResponse({ status: "error" }); }
  const ss = SPREADSHEET_ID ? SpreadsheetApp.openById(SPREADSHEET_ID) : SpreadsheetApp.getActiveSpreadsheet();
  const action = postData.action;

  try {
    if (action === "login") {
      const users = getSheetData(ss.getSheetByName("Users"));
      const user = users.find(u => u.email === postData.email && u.password_hash === postData.password);
      if (user) return contentResponse({ status: "success", user: { id: user.id, name: user.name, email: user.email, role: user.role } });
      else return contentResponse({ status: "error", message: "Gagal Login!" });
    }

    if (action === "savePost") {
      const sheet = ss.getSheetByName("Posts");
      const post = postData.post;
      // Validasi keamanan: Jika user bukan admin, status paksa jadi draft
      if (postData.user_role !== 'admin') {
         post.status = 'draft';
      }
      
      let finalImageUrl = post.thumbnail_url;
      if (post.image_base64 && post.image_filename) finalImageUrl = uploadImageToDrive(post.image_base64, post.image_filename);

      const existingIndex = getSheetData(sheet).findIndex(p => p.id === post.id);
      const ts = new Date().toISOString();

      if (existingIndex !== -1) {
        const row = existingIndex + 2;
        sheet.getRange(row, 2, 1, 7).setValues([[post.title, post.slug, post.content, finalImageUrl, post.category_id, post.author_id, post.status]]);
        sheet.getRange(row, 10).setValue(ts);
      } else {
        sheet.appendRow([post.id, post.title, post.slug, post.content, finalImageUrl || "", post.category_id, post.author_id, post.status, ts, ts]);
      }
      return contentResponse({ status: "success" });
    }

    if (action === "deletePost") {
      const sheet = ss.getSheetByName("Posts");
      const idx = getSheetData(sheet).findIndex(p => p.id === postData.id);
      if (idx !== -1) sheet.deleteRow(idx + 2);
      return contentResponse({ status: "success" });
    }
    
    // SAVE SETTINGS
    if (action === "saveSettings") {
      const sheet = ss.getSheetByName("Settings");
      const settings = postData.settings;
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        const key = rows[i][0];
        if (settings[key] !== undefined) sheet.getRange(i + 1, 2).setValue(settings[key]);
      }
      return contentResponse({ status: "success" });
    }
    
    // Kategori & Users skip for brevity... sama seperti versi sebelumnya
    return contentResponse({ status: "success", message: "Aksi sukses divalidasi API." });
  } catch (err) {
    return contentResponse({ status: "error", message: err.toString() });
  }
}

// FUNGSI UNGGUH GAMBAR KE GOOGLE DRIVE
function uploadImageToDrive(base64Data, fileName) {
  try {
    const rawData = Utilities.base64Decode(base64Data.split(",")[1]);
    const blob = Utilities.newBlob(rawData, "image/png", fileName);
    let folder;
    const folders = DriveApp.getFoldersByName("Media_Blog");
    if (folders.hasNext()) folder = folders.next();
    else folder = DriveApp.createFolder("Media_Blog");
    
    const file = folder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return "https://lh3.googleusercontent.com/d/" + file.getId();
  } catch (e) {
    return ""; 
  }
}

function getSheetData(sheet) {
  if (!sheet) return [];
  const vals = sheet.getDataRange().getValues();
  const headers = vals[0];
  return vals.slice(1).map(row => headers.reduce((obj, h, i) => { obj[h] = row[i]; return obj; }, {}));
}

function getSettingsMap(sheet) {
  if (!sheet) return {};
  const vals = sheet.getDataRange().getValues();
  return vals.slice(1).reduce((map, row) => { map[row[0]] = row[1]; return map; }, {});
}

function contentResponse(data) {
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

2 Deploy & Selesai

  1. Jalankan fungsi setupDatabase di Apps Script (Pilih fungsi di menu atas, klik Jalankan).
  2. Klik Terapkan (Deploy) > Terapkan Baru. Jenis: Aplikasi Web.
  3. Setel Akses ke: Siapa Saja (Anyone). Salin URL Aplikasi Web. Tempel di kolom atas. Selesai!
